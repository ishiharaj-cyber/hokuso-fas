function doPost(e) {
  var sheet = SpreadsheetApp.openById("1UOmKA8c2W84lKa1b2ol5S4KHf1caD8df2ioCecg1xvg").getSheetByName("Web予約");
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([new Date(), data.groupCount, data.participantCount, data.asset, data.eventName, "", data.plannerName]);
  return ContentService.createTextOutput("ok");
}

function doGet(e) {
  var sheet = SpreadsheetApp.openById("1UOmKA8c2W84lKa1b2ol5S4KHf1caD8df2ioCecg1xvg").getSheetByName("Web予約");
  var eventName = (e.parameter.event || "").toString().trim();
  var data = sheet.getDataRange().getValues();
  var totalGroups = 0;
  var totalPeople = 0;
  var nameGroups = {};
  for (var i = 1; i < data.length; i++) {
    var cellValue = (data[i][4] || "").toString().trim();
    if (cellValue === eventName || cellValue.indexOf(eventName) >= 0 || eventName.indexOf(cellValue) >= 0) {
      var groups = Number(data[i][1] || 0);
      totalGroups = totalGroups + groups;
      totalPeople = totalPeople + Number(data[i][2] || 0);
      var name = (data[i][6] || "").toString().trim();
      if (name) {
        nameGroups[name] = (nameGroups[name] || 0) + groups;
      }
    }
  }
  var planners = [];
  for (var n in nameGroups) {
    planners.push({name: n, groups: nameGroups[n]});
  }
  var result = JSON.stringify({groups: totalGroups, people: totalPeople, planners: planners});
  return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON);
}
